instance TextShow c => TextShow1 (K1 i c) where

-- | /Since: 2.1.2/
instance TextShow1 UChar where
    liftShowbPrec _ _ p (UChar c) = showbParen (p > appPrec) $
           "UChar "    <> singleton '{'
        <> "uChar# = " <> showb (C# c)
        <> singleton '}'
